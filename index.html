var curStep = 0, curSub = 0, curInstr = 0;

var doneSteps = [], doneSubs = {};


function isDoneStep(i) { for (var j = 0; j < doneSteps.length; j++) { if (doneSteps[j] === i) return true; } return false; }

function isDoneSub(si, sj) { if (!doneSubs[si]) return false; for (var k = 0; k < doneSubs[si].length; k++) { if (doneSubs[si][k] === sj) return true; } return false; }

function markDoneStep(i) { if (!isDoneStep(i)) doneSteps.push(i); }

function markDoneSub(si, sj) { if (!doneSubs[si]) doneSubs[si] = []; if (!isDoneSub(si, sj)) doneSubs[si].push(sj); }


function aStep() { return steps[curStep]; }

function aSub() { var s = aStep(); return s.subSteps ? s.subSteps[curSub] : null; }

function aInstrs() { var sub = aSub(); return sub ? sub.instructions : aStep().instructions; }

function aTip() { var sub = aSub(); return (sub !== null && sub !== undefined) ? sub.tip : aStep().tip; }

function aMockup() { var sub = aSub(); return sub ? sub.getMockup(curInstr) : aStep().getMockup(curInstr); }


function goToStep(i) { curStep = i; curSub = 0; curInstr = 0; activeSFPanel = null; render(); }

function goToSub(j) { curSub = j; curInstr = 0; activeSFPanel = null; render(); }


function nextInstr() {

  var n = curInstr + 1;

  if (n < aInstrs().length) { curInstr = n; switchMockup(curInstr); }

  else { advance(); }

}


function advance() {

  var s = aStep();

  if (s.subSteps) {

    markDoneSub(curStep, curSub);

    if (curSub < s.subSteps.length - 1) { curSub++; curInstr = 0; render(); }

    else { markDoneStep(curStep); curStep++; curSub = 0; curInstr = 0; render(); }

  } else {

    markDoneStep(curStep); curStep++; curInstr = 0; render();

  }

}

  } else {

    if (curStep > 0) { curStep--; var p = steps[curStep]; curSub = p.subSteps ? p.subSteps.length - 1 : 0; curInstr = 0; render(); }

  }

}

function updateInlineBtn() {

  var btn = document.getElementById('inlineNext');

  if (!btn) return;

  var instrs = aInstrs(), isLast = curInstr >= instrs.length - 1;

  var s = aStep(), lastSub = s.subSteps ? curSub >= s.subSteps.length - 1 : true, lastStep = curStep >= steps.length - 1;

  btn.textContent = (isLast && lastSub && lastStep) ? 'Finish' : isLast ? 'Next Section' : 'Next';

}

function renderNav() {

  var nav = document.getElementById('stepNav');

  var html = '';

  for (var i = 0; i < steps.length; i++) {

    var s = steps[i];

    var isActive = i === curStep, isDone = isDoneStep(i);

    html += '<div class="nav-item' + (isActive ? ' active' : '') + (isDone && !isActive ? ' completed' : '') + '" onclick="goToStep(' + i + ')">';

    html += '<div class="nav-num">' + (isDone && !isActive ? '&#x2713;' : (i + 1)) + '</div>';

    html += '<span>' + s.title + '</span></div>';

    if (s.subSteps && isActive) {

      html += '<div class="sub-nav">';

      for (var j = 0; j < s.subSteps.length; j++) {

        var sub = s.subSteps[j], sa = j === curSub, sd = isDoneSub(i, j);

        html += '<div class="sub-nav-item' + (sa ? ' active' : '') + (sd && !sa ? ' completed' : '') + '" onclick="goToSub(' + j + ')">';

        html += '<div class="sub-nav-dot">' + (sd && !sa ? '&#x2713;' : sub.icon) + '</div>';

        html += '<span>' + sub.title + '</span></div>';

      }

      html += '</div>';

    }

  }

  nav.innerHTML = html;

}
